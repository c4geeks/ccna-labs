! ===========================================================================
! R2 static-routing-ipv4 lab  (router, c7200 / IOS 15.2)
! ComputingForGeeks CCNA 200-301 lab: static-routing-ipv4
! https://computingforgeeks.com/cisco-ipv4-static-routes-configuration/
!
! Two links to R1: primary 10.0.12.0/30 (Gi0/0), backup 10.0.13.0/30 (Gi1/0).
! R2 reaches R1's LAN 192.168.1.0/24 via a primary static (AD 1) and a floating
! static (AD 200) over the backup, so the return path also fails over when the
! primary link is shut. Run the matching R1 file on the other router.
! ===========================================================================
enable
configure terminal
hostname R2
interface Loopback1
 ip address 192.168.2.1 255.255.255.0
 exit
interface GigabitEthernet0/0
 ip address 10.0.12.2 255.255.255.252
 no shutdown
 exit
interface GigabitEthernet1/0
 ip address 10.0.13.2 255.255.255.252
 no shutdown
 exit
ip route 192.168.1.0 255.255.255.0 10.0.12.1
ip route 192.168.1.0 255.255.255.0 10.0.13.1 200
end
write memory
